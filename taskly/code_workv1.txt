def read_sheet(range_):
    try:
        result = service.spreadsheets().values().get(
            spreadsheetId=SHEET_ID,
            range=range_
        ).execute()
        values = result.get("values", [])
        if not values:
            return pd.DataFrame()
        
        # Ä°lk satÄ±r baÅŸlÄ±k satÄ±rÄ±
        headers = values[0] if values else []
        data_rows = values[1:] if len(values) > 1 else []
        
        # EÄŸer veri yoksa boÅŸ DataFrame dÃ¶ndÃ¼r
        if not data_rows:
            return pd.DataFrame(columns=headers if headers else ["ID", "BaÅŸlÄ±k", "Periyot", "BaÅŸlangÄ±Ã§ Tarihi", "Sonraki Tarih", "TamamlandÄ±"])
        
        # Her satÄ±rÄ± aynÄ± uzunlukta yap (eksik sÃ¼tunlarÄ± boÅŸ string ile doldur)
        max_cols = len(headers) if headers else 6
        normalized_rows = []
        for row in data_rows:
            # SatÄ±rÄ± normalize et
            normalized_row = row[:]  # Kopyala
            while len(normalized_row) < max_cols:
                normalized_row.append("")
            # Fazla sÃ¼tunlarÄ± kes
            normalized_row = normalized_row[:max_cols]
            normalized_rows.append(normalized_row)
        
        # DataFrame oluÅŸtur
        if headers:
            df = pd.DataFrame(normalized_rows, columns=headers)
        else:
            df = pd.DataFrame(normalized_rows, columns=["ID", "BaÅŸlÄ±k", "Periyot", "BaÅŸlangÄ±Ã§ Tarihi", "Sonraki Tarih", "TamamlandÄ±"])
        
        return df
        
    except Exception as e:
        st.error(f"Google Sheets'ten okunamadÄ±: {e}")
        return pd.DataFrame()

def write_sheet(range_name, values):
    try:
        # NaN deÄŸerlerini boÅŸ string ile deÄŸiÅŸtir
        clean_values = []
        for row in values:
            clean_row = []
            for cell in row:
                if pd.isna(cell) or cell == 'NaN' or str(cell) == 'nan':
                    clean_row.append("")
                else:
                    clean_row.append(str(cell))
            clean_values.append(clean_row)
        
        service.spreadsheets().values().update(
            spreadsheetId=SHEET_ID,
            range=range_name,
            valueInputOption="USER_ENTERED",
            body={"values": clean_values}
        ).execute()
    except Exception as e:
        st.error(f"Google Sheets'e yazÄ±lamadÄ±: {e}")
        raise

def save_tasks_to_sheet(df):
    # DataFrame'deki NaN deÄŸerlerini boÅŸ string ile deÄŸiÅŸtir
    df_clean = df.fillna("")
    values = [df_clean.columns.tolist()] + df_clean.values.tolist()
    write_sheet(f"{SHEET_NAME}!A1", values)

@st.cache_data(ttl=60)

    df = read_sheet(f"{SHEET_NAME}!A1:Z1000")

    required_columns = ["ID", "BaÅŸlÄ±k", "Periyot", "BaÅŸlangÄ±Ã§ Tarihi", "Sonraki Tarih", "TamamlandÄ±"]

def send_email(to_email, subject, body):
    try:
        # E-mail ayarlarÄ±nÄ± secrets'tan al
        smtp_server = st.secrets.get("email", {}).get("smtp_server", "smtp.gmail.com")
        smtp_port = st.secrets.get("email", {}).get("smtp_port", 587)
        from_email = st.secrets.get("email", {}).get("from_email", "")
        password = st.secrets.get("email", {}).get("password", "")
        
        if not from_email or not password:
            st.warning("E-mail ayarlarÄ± eksik. Bildirim gÃ¶nderilemedi.")
            return False
        
        msg = MIMEMultipart()
        msg['From'] = from_email
        msg['To'] = to_email
        msg['Subject'] = subject
        msg.attach(MIMEText(body, 'plain', 'utf-8'))
        
        server = smtplib.SMTP(smtp_server, smtp_port)
        server.starttls()
        server.login(from_email, password)
        server.send_message(msg)
        server.quit()
        return True
    except Exception as e:
        st.error(f"E-mail gÃ¶nderme hatasÄ±: {e}")
        return False
